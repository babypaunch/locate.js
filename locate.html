locate.js - Sample

Example 1. locate(url)
$(function(){ $("#locate1").click(function(){ locate("http://www.google.com"); }); })

Example 2-1. locate(url, param)
$(function(){ $("#locate2-1").click(function(){ locate("http://www.google.com", "p1=111&p2=222"); }); })

Example 2-2. locate(url, param)
$(function(){ $("#locate2-2").click(function(){ locate($("#urlString2-2").val(), $("#form2-2").serialize()); }); })




Example 3. locate(url, $form)
$(function(){ $("#locate3").click(function(){ locate($("#urlString3").val(), $("#form3")); }); })

Example 4-1. locate(url, json)
$(function(){ $("#locate4-1").click(function(){ locate($("#urlString4-1").val(), { type: "get" , sendType: "text" , returnType: "text" , callback: "locate4_1" }); }); })

Example 4-2. locate(url, json)
$(function(){ $("#locate4-2").click(function(){ locate($("#urlString5").val(), { param: $("#form4-2").seiralize() , type: "get" , sendType: "text" , returnType: "text" , callback: "locate4_2" }); }); })
