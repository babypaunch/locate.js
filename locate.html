locate.js - Sample

Example 1. locate(url) - page move
$(function(){ $("#locate1").click(function(){ locate("http://www.google.com"); }); })

Example 2-1. locate(url, param) - page move + param
$(function(){ $("#locate2-1").click(function(){ locate("http://www.google.com", "p1=111&p2=222"); }); })

Example 2-2. locate(url, param) - page move + serialized param
$(function(){ $("#locate2-2").click(function(){ locate($("#urlString2-2").val(), $("#form2-2").serialize()); }); })




Example 3. locate(url, $form) - page submit + form
$(function(){ $("#locate3").click(function(){ locate($("#urlString3").val(), $("#form3")); }); })

Example 4-1. locate(url, json) - jsonp + no param(default empty string) + custom loading
$(function(){ $("#locate4-1").click(function(){ locate($("#urlString4-1").val(), { param: $("#form4-1").seiralize() , type: "get" , returnType: "jsonp" , done: function(data){ console.log("[locate4_1][done]", data); } , fail: function(data){ console.log("[locate4_1][fail]", data); } , loading: function(){ alert("LOADING"); } , unloading: function(){ alert("UNLOADING"); } }); }); })

Example 4-2. locate(url, json) - jsonp + serialized form(has no input[type='file'])
$(function(){ $("#locate4-2").click(function(){ locate($("#urlString4-2").val(), { param: $("#form4-2") , type: "get" , returnType: "jsonp" , done: function(data){ console.log("[locate4_2][done]", data); } , fail: function(data){ console.log("[locate4_2][fail]", data); } }); }); })




Example 4-3. locate(url, json) - jsonp + json param
$(function(){ $("#locate4-3").click(function(){ locate($("#urlString4-3").val(), { param: {a: "12", b: "34"} , type: "get" , returnType: "jsonp" , done: function(data){ console.log("[locate4_3][done]", data); } , fail: function(data){ console.log("[locate4_3][fail]", data); } }); }); })

Example 4-4. locate(url, json) - jsonp + form(has input[type='file'])
$(function(){ $("#locate4-4").click(function(){ locate($("#urlString4-4").val(), { param: $("#form4-4") , type: "get" , returnType: "jsonp" , done: function(data){ console.log("[locate4_4][done]", data); } , fail: function(data){ console.log("[locate4_4][fail]", data); } }); }); })



Example 5. route/load similar
$(function(){ $("#loadA").click(function(){ $("#loaded").locate("a.html #login"); }); $("#loadB").click(function(){ $("#loaded").locate({ url: "b.html" }); }); $("#errorC").click(function(){ $("#loaded").locate("c.html", function(data){ console.log(data); }, true); }); $("#errorD").click(function(){ $("#loaded").locate({ url: "d.html" , done: function(data){ console.log(data); } , fail: function(){ alert("ERROR"); } }); }); })
